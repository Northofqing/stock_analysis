### 5.1 AuctionPulse (~80 行)

```rust
// src/auction_pulse.rs (复用现有 push_templates.rs:4130 路径)
use crate::push_templates::load_auction_volume_snapshot_real;
use crate::monitor::monitor_loop::now_china;

pub struct AuctionPulse {
    last_emit_window: HashMap<AuctionWindow, Instant>,
    seen_volumes: HashSet<(String, NaiveTime)>,    // dedup 同一 (code, time)
}

#[derive(Hash, Eq, PartialEq, Clone, Copy)]
pub enum AuctionWindow { O915, O920, O925 }   // 3 个窗口

impl AuctionPulse {
    /// 9:15-9:25 每 30s tick (async)
    pub async fn tick(&mut self, now: NaiveTime) -> Vec<DomainEvent> {
        let mut out = Vec::new();

        // 1. 9:25 准点推一次完整汇总 (主推)
        for window in [AuctionWindow::O915, O920, O925] {
            let target = window.to_naive_time();
            // NaiveTime 减法返回 i32 秒数 (chrono's SignedDuration), 与目标 ±30s 比较
            let delta_secs = (now.num_seconds_from_midnight() as i64
                              - target.num_seconds_from_midnight() as i64).abs();
            if delta_secs <= 30 {
                if self.should_emit(window) {
                    let hhmm = format!("{:02}:{:02}", target.hour(), target.minute());
                    let snapshot = load_auction_volume_snapshot_real(&hhmm);
                    out.push(DomainEvent::auction_pulse(window, snapshot));
                    self.last_emit_window.insert(window, Instant::now());
                }
            }
        }

        out
    }
}
```

```rust
use crate::market_analyzer::MarketAnalyzer;

pub struct LimitBoardWatcher {
    last_snapshot_hash: Option<u64>,    // L4 dedup 关键
    last_emit: Instant,
}

impl LimitBoardWatcher {
    /// 每 30s tick (async)
    pub async fn tick(&mut self) -> Option<DomainEvent> {
        // 复用 MarketAnalyzer::get_limit_up_stocks (3-host 降级 + ST 过滤 + 北交所过滤)
        let snapshot: Vec<LimitUpItem> = MarketAnalyzer::get_limit_up_stocks().await;
        // 必须用 Vec 而非 HashMap 排列 (HashMap 在多线程下迭代顺序随机, hash 不确定)
        let mut sorted = snapshot.clone();
        sorted.sort_by(|a, b| a.code.cmp(&b.code));
        let hash = hash_of(&sorted);
        if Some(hash) == self.last_snapshot_hash {
            return None;   // 内容未变 → 不推 (降噪)
        }
        self.last_snapshot_hash = Some(hash);
        Some(DomainEvent::limit_board_update(snapshot))
    }
}
```

```rust
pub async fn try_pre_open_reset_br021(now_local: NaiveTime) -> Option<AccountMode> {
    if now_local < NaiveTime::from_hms_opt(8, 30, 0).unwrap()
        || now_local >= NaiveTime::from_hms_opt(8, 31, 0).unwrap() {
```
