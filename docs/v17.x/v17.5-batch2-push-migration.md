### 5.1 AuctionPulse (~80 行)

```rust
// src/auction_pulse.rs
pub struct AuctionPulse {
    last_emit_window: HashMap<AuctionWindow, Instant>,
    seen_volumes: HashSet<(String, NaiveTime)>,    // dedup 同一 (code, time)
}

#[derive(Hash, Eq, PartialEq, Clone, Copy)]
pub enum AuctionWindow { O915, O920, O925 }   // 3 个窗口

impl AuctionPulse {
    /// 9:15-9:25 每 30s tick
    pub fn tick(&mut self, now: NaiveTime) -> Vec<DomainEvent> {
        let mut out = Vec::new();

        // 1. 9:25 准点推一次完整汇总 (主推)
        for window in [AuctionWindow::O915, O920, O925] {
            let target = window.to_naive_time();
            if (now - target).abs() < Duration::from_secs(30) {
                if self.should_emit(window) {
                    let summary = auction_provider::read_summary(window).await;
                    out.push(DomainEvent::auction_pulse(window, summary));
                    self.last_emit_window.insert(window, Instant::now());
                }
            }
        }

        // 2. 9:20 repush: 复用同窗口
        out
    }
}
```

**关键决策**:
- AuctionVolume + AuctionRepush 合并到 AuctionPulse (同窗口模式, 共用 dedup)
- 9:20 repush 是 AuctionPulse 的"重复触发"模式,不是另一个 EventKind
- L4 dedup 关键字段: `(window, sorted_by_volume)` hash

```rust
impl LimitBoardWatcher {
    /// 每 30s tick
    pub fn tick(&mut self) -> Option<DomainEvent> {
        let snapshot = limit_board_provider::scan().await;
```

#### 5.10.2 BR-021 修复方案

```rust
// src/push_l5/br021.rs  (NEW, ~80 行)

use chrono::{Local, NaiveTime, Weekday};
use crate::portfolio::state::AccountMode;

pub struct Br021Guard {
    is_frozen: bool,
    frozen_since: Option<NaiveDate>,
}

impl Br021Guard {
    /// 8:30 PreOpen cron 调用: 检查并重置
    pub async fn pre_open_reset(&mut self) -> Option<DomainEvent> {
        if !self.is_frozen {
            return None;
        }

        // 盘前重置: Frozen → 回 Real (默认)
        log::warn!("[BR-021] Frozen 状态检测 → 盘前自动重置为 Real");
        self.is_frozen = false;
        self.frozen_since = None;

        Some(DomainEvent::account_mode_changed(
            AccountMode::Real,
            "BR-021 盘前重置".to_string()
        ))
    }

    /// AccountMode cron 调用: 切到 Frozen 时记录
    pub fn on_account_mode_changed(&mut self, new_mode: AccountMode) {
        match new_mode {
            AccountMode::Frozen => {
                self.is_frozen = true;
                self.frozen_since = Some(Local::now().date_naive());
                log::warn!("[BR-021] AccountMode=Frozen 已记录,等待 {} 盘前重置",
                    Local::now().date_naive().succ_opt().unwrap());
            }
            _ => {
                // 用户主动切走 → 清状态
                self.is_frozen = false;
                self.frozen_since = None;
            }
        }
    }
}
```

```rust

    }
}
```
