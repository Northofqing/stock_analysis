## 5. 设计 — 3 模块详细

### 5.1 NewsAggregator (~200 行)

```rust
// src/news_aggregator.rs

use std::collections::HashMap;
use std::time::{Duration, Instant};
use chrono::NaiveTime;
use crate::opportunity::news_ranker::{RankedNews, NewsWindow};

pub struct NewsAggregator {
    last_emit_window: HashMap<NewsWindow, Instant>,
    seen_critical_ids: HashSet<String>,    // dedup critical
    dedup_window: Duration,                 // 默认 5 min
}

#[derive(Hash, Eq, PartialEq, Clone, Copy)]
pub enum NewsWindow { O930, O1130, O1300, O1500 }

impl NewsAggregator {
    pub fn new(dedup_window_secs: u64) -> Self {
        Self {
            last_emit_window: HashMap::new(),
            seen_critical_ids: HashSet::new(),
            dedup_window: Duration::from_secs(dedup_window_secs),
        }
    }

    /// monitor_loop 每 30 秒调一次
    pub fn tick(&mut self, now: NaiveTime) -> Vec<DomainEvent> {
        let mut out = Vec::new();

        // 1. 事件驱动: 读 news_audit 新行, score≥80 即时推
        for news in news_ranker::read_new_lines().await {
            if news.score >= 80 && !self.seen_critical_ids.contains(&news.id) {
                self.seen_critical_ids.insert(news.id.clone());
                out.push(DomainEvent::news_flash_critical(news));
            }
        }

        // 2. 4 时段聚合: 9:30/11:30/13:00/15:00 ±1min 触发
        for window in [NewsWindow::O930, O1130, O1300, O1500] {
            let target = window.to_naive_time();
            if (now - target).abs() < Duration::from_secs(60) {
                let should_emit = self.last_emit_window
                    .get(&window)
                    .map(|t| t.elapsed() > self.dedup_window)
                    .unwrap_or(true);
                if should_emit {
                    self.last_emit_window.insert(window, Instant::now());
                    let top3 = news_ranker::top_n(window, 3).await;
                    out.push(DomainEvent::news_flash_aggregated(window, top3));
                }
            }
        }

        out
    }
}
```

### 5.2 PaperReviewer (~250 行)

```rust
// src/paper_reviewer.rs

use crate::portfolio::store::{read_today_paper_trades, read_execution_tracking};
use crate::broker::QuoteProvider;

pub struct PaperReviewer {
    quote_provider: Arc<dyn QuoteProvider>,
}

pub struct PaperSnapshot {
    pub top3: Vec<HoldingPerf>,         // 涨最多 Top 3
    pub bottom3: Vec<HoldingPerf>,      // 跌最多 Top 3
    pub current_holdings: Vec<HoldingValuation>,  // 持仓 + cost + current + pnl
}

pub struct FullReview {
    pub snapshot: PaperSnapshot,
    pub winrate_by_kind: HashMap<PushKind, f64>,   // 按 push.kind 拆
    pub real_alpha: f64,                            // 对标市场
    pub tomorrow_watchlist: Vec<WatchItem>,         // 明日观察 Top 5
    pub sop: String,                                // 明日 SOP (竞价/开盘/拉升/跳水)
    pub factor_report: Option<String>,              // 周一才有
}

impl PaperReviewer {
    /// 13:00 由 ReviewCron 调
    pub async fn snapshot_noon(&self) -> DomainEvent {
        let today = read_today_paper_trades();
        let current_holdings = self.valuation_holdings().await;

        let mut all = today.clone();
        all.extend(current_holdings.clone());
        let mut sorted: Vec<_> = all.into_iter().collect();
        sorted.sort_by(|a, b| b.pnl_pct.partial_cmp(&a.pnl_pct).unwrap());

        let snapshot = PaperSnapshot {
            top3: sorted.iter().take(3).cloned().collect(),
            bottom3: sorted.iter().rev().take(3).cloned().collect(),
            current_holdings,
        };

        DomainEvent::paper_review_snapshot(snapshot)
    }

    /// 15:35 由 ReviewCron 调
    pub async fn full_review(&self) -> DomainEvent {
        let snapshot = self.snapshot_noon().await;
        let winrate_by_kind = self.calc_winrate_by_kind().await;
        let real_alpha = real_alpha::compute(&winrate_by_kind).await;
        let tomorrow_watchlist = tomorrow_watchlist::generate_top5().await;
        let sop = sop::daily().await;
        let factor_report = if chrono::Local::now().weekday() == Weekday::Mon {
            Some(factor_report::weekly().await)
        } else {
            None
        };

        DomainEvent::paper_review_daily(FullReview {
            snapshot,
            winrate_by_kind,
            real_alpha,
            tomorrow_watchlist,
            sop,
            factor_report,
        })
    }

    /// 持仓实时估值 — quote_provider 失败 graceful
    async fn valuation_holdings(&self) -> Vec<HoldingValuation> {
        let mut out = Vec::new();
        for h in portfolio::current_holdings() {
            let price = self.quote_provider.get_quote_price(&h.code);
            if price == 0.0 {
                log::warn!("[持仓估值] {} quote_provider 返回 0.0, graceful 降级", h.code);
            }
            out.push(HoldingValuation {
                code: h.code,
                name: h.name,
                cost: h.cost,
                current: price,
                pnl_pct: if h.cost > 0.0 { (price - h.cost) / h.cost * 100.0 } else { 0.0 },
            });
        }
        out
    }

    /// 按 push.kind 拆胜率
    async fn calc_winrate_by_kind(&self) -> HashMap<PushKind, f64> {
        let mut out = HashMap::new();
        for kind in [PushKind::NewsFlashCritical, NewsFlashAggregated,
                     PaperReviewSnapshot, PaperReviewDaily,
                     // ... 其他 push.kind
                    ] {
            let stats = history_query::winrate(kind, 30).await;  // 30 天窗口
            out.insert(kind, stats.winrate);
        }
        out
    }
}
```
